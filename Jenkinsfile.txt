pipeline {
    agent any

    environment {
        DOCKER_HUB_USER = 'your-dockerhub-username'
        IMAGE_NAME      = 'httpd-web-app'
        IMAGE_TAG       = "${BUILD_NUMBER}" // Using build number for unique versioning
        REGISTRY_CRED   = 'docker-hub-credentials'
    }

    stages {
        stage('Fetch Source Code') {
            steps {
                // Pulls latest changes from GitHub
                checkout scm
            }
        }

        stage('Build Docker Image') {
            steps {
                script {
                    echo "Building Docker image: ${DOCKER_HUB_USER}/${IMAGE_NAME}:${IMAGE_TAG}"
                    // Builds the image using the local Dockerfile
                    customImage = docker.build("${DOCKER_HUB_USER}/${IMAGE_NAME}:${IMAGE_TAG}")
                    // Also tag it as latest
                    customImage.tag("latest")
                }
            }
        }

        stage('Push Image to Docker Hub') {
            steps {
                script {
                    // Authenticates and pushes images using credentials configured in Jenkins
                    docker.withRegistry('https://index.docker.io/v1/', REGISTRY_CRED) {
                        customImage.push()
                        customImage.push("latest")
                    }
                }
            }
        }

        stage('Deploy to Minikube') {
            steps {
                script {
                    echo "Updating Minikube deployment with new image version..."
                    
                    // Updates the image configuration inside the deployment manifest directly
                    sh "sed -i 's|image: .*|image: ${DOCKER_HUB_USER}/${IMAGE_NAME}:${IMAGE_TAG}|g' k8s.yaml"
                    
                    // Applies the updated manifest to the local cluster
                    sh "kubectl apply -f k8s.yaml"
                    
                    // Forces a rolling restart ensuring pods pull the fresh image
                    sh "kubectl rollout restart deployment/httpd-web-deployment"
                }
            }
        }
    }

    post {
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed. Check build logs for details.'
        }
    }
}
